JavaScript (Lógica Completa) Aqui injetamos a inteligência: criação automática de pastilhas, detecção de colisão e o movimento do fantasma. JavaScript const board = document.getElementById("game-board"); const pacman = document.getElementById("pacman"); const ghost = document.getElementById("ghost"); const scoreElement = document.getElementById("score"); let pacmanX = 0, pacmanY = 0; let ghostX = 360, ghostY = 360; let score = 0; const step = 40; // Tamanho da grade // 1. Gerar pastilhas no mapa function createDots() { for (let y = 0; y < 400; y += step) { for (let x = 0; x < 400; x += step) { if (x === 0 && y === 0) continue; // Pula a posição inicial do Pac-Man let dot = document.createElement("div"); dot.className = "dot"; dot.style.left = x + "px"; dot.style.top = y + "px"; board.appendChild(dot); } } } // 2. Movimento do Pac-Man e Colisões document.addEventListener("keydown", (e) => { let oldX = pacmanX; let oldY = pacmanY; if (e.key === "ArrowUp" && pacmanY > 0) { pacmanY -= step; pacman.style.transform = "rotate(270deg)"; } if (e.key === "ArrowDown" && pacmanY < 360) { pacmanY += step; pacman.style.transform = "rotate(90deg)"; } if (e.key === "ArrowLeft" && pacmanX > 0) { pacmanX -= step; pacman.style.transform = "rotate(180deg)"; } if (e.key === "ArrowRight" && pacmanX < 360) { pacmanX += step; pacman.style.transform = "rotate(0deg)"; } pacman.style.left = pacmanX + "px"; pacman.style.top = pacmanY + "px"; checkEat(); }); // 3. Sistema de Comer function checkEat() { const dots = document.querySelectorAll(".dot"); dots.forEach(dot => { if (parseInt(dot.style.left) === pacmanX && parseInt(dot.style.top) === pacmanY) { dot.remove(); score += 10; scoreElement.innerText = score; } }); } // 4. Inteligência do Fantasma (Perseguição básica) function moveGhost() { if (ghostX < pacmanX) ghostX += step; else if (ghostX > pacmanX) ghostX -= step; else if (ghostY < pacmanY) ghostY += step; else if (ghostY > pacmanY) ghostY -= step; ghost.style.left = ghostX + "px"; ghost.style.top = ghostY + "px"; if (ghostX === pacmanX && ghostY === pacmanY) { alert("GAME OVER! Pontuação: " + score); location.reload(); } } // Inicialização createDots(); setInterval(moveGhost, 400); // Fantasma se move a cada 0.4s
